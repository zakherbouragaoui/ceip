🔑These contracts are your quality guarantee. If Gemma 4 returns output that doesn't match this schema, it is rejected and retried automatically — no bad data ever reaches a user.

Classify papers with Gemma 4 (first structured AI use)

# src/intelligence/classify.py import requests, json from contracts.evidence import PaperClassification PROMPT = """Extract structured metadata from this abstract. Return ONLY valid JSON matching this schema exactly: { "intervention_type": one of ["habitat_restoration","species_reintro", "protected_area","invasive_control","captive_breeding", "community_mgmt","policy","monitoring","other"], "species_group": one of ["mammals","birds","reptiles","fish", "invertebrates","plants","fungi","general"], "outcome": one of ["positive","negative","mixed","unclear"], "geography": "country or region or global", "has_quantitative_result": true or false } Abstract: {abstract}""" def classify_paper(abstract: str, max_retries=3) -> PaperClassification | None: for attempt in range(max_retries): try: r = requests.post("http://localhost:11434/api/generate", json={ "model": "gemma4:e4b", "prompt": PROMPT.format(abstract=abstract), "stream": False, "format": "json" # Gemma 4 native JSON mode }) return PaperClassification(**json.loads(r.json()['response'])) except: continue return None
ℹ️Gemma 4's native "format": "json" parameter enforces JSON output at the model level — a major advantage over older models that required prompt tricks to get structured output.

# src/living/ingestion_agent.py from datetime import datetime, timedelta import httpx, asyncio from loguru import logger CONSERVATION_CONCEPTS = [ "C149923435", # Conservation biology (OpenAlex concept ID) "C2780878898", # Biodiversity "C2993671", # Wildlife conservation "C100360200", # Habitat conservation ] async def fetch_new_papers(days_ago=7): since = (datetime.now() - timedelta(days=days_ago)).strftime("%Y-%m-%d") async with httpx.AsyncClient() as client: for concept_id in CONSERVATION_CONCEPTS: r = await client.get("https://api.openalex.org/works", params={ "filter": f"concepts.id:{concept_id},from_publication_date:{since}", "per-page": 200, "select": "id,title,abstract_inverted_index,doi,publication_year" }) yield from r.json().get('results', []) async def run_weekly_ingestion(): logger.info("Starting weekly ingestion run") new, skipped = 0, 0 async for paper in fetch_new_papers(): doi = paper.get('doi') if doi and not db.paper_exists(doi): classification = classify_paper(reconstruct_abstract(paper)) if classification and classification.is_conservation_relevant: db.queue_for_embedding(doi, paper, classification) new += 1 else: skipped += 1 logger.info(f"Ingestion: {new} new, {skipped} skipped/duplicate")

2
Chunk documents and run batch embedding overnight
Split papers into 400-token chunks with 50-token overlap. Embed CE interventions whole (they're short). Run the full batch overnight — 15,000 papers × ~3 chunks = ~45,000 embeddings via nomic-embed-text on Metal GPU.
from langchain.text_splitter import RecursiveCharacterTextSplitter from tqdm import tqdm import requests splitter = RecursiveCharacterTextSplitter(chunk_size=400, chunk_overlap=50) def embed(text: str) -> list[float]: r = requests.post("http://localhost:11434/api/embeddings", json={"model": "nomic-embed-text", "prompt": text}) return r.json()['embedding'] def embed_papers_batch(batch_size=50): papers = db.get_unembedded_papers() for paper in tqdm(papers, desc="Embedding"): chunks = splitter.split_text(paper.abstract) for i, chunk in enumerate(chunks): papers_col.add( documents=[chunk], embeddings=[embed(chunk)], ids=[f"{paper.id}_c{i}"], metadatas=[{"paper_id": paper.id, "year": paper.year, "intervention_type": paper.intervention_type}] ) db.mark_embedded(paper.id) # CE interventions: combine action + category + synopsis for richer embedding def embed_ce_interventions(): for ce in db.get_all_interventions(): text = f"{ce.action} [{ce.ce_category}] {ce.synopsis}" ce_col.add(documents=[text], embeddings=[embed(text)], ids=[f"ce_{ce.id}"], metadatas=[{"ce_id": ce.id, "effectiveness": ce.effectiveness or 0}])

Intelligence Layer · Weeks 9–12
Harness-First RAG Engine
3 weeks 6 steps 6 tests
The harness principle applied fully: Query routing, Pydantic-enforced outputs, retry logic, and validation happen BEFORE any response is formed. The LLM is one component — not the system. Switch Gemma 4 for any future model by changing one config line.
Steps
1
Build the query router — not everything goes to the LLM
Three query types; only complex ones invoke the full RAG pipeline. Simple queries return fast, accurate answers from structured data without ever touching the LLM.
# src/intelligence/router.py from enum import Enum class QueryType(Enum): FACTUAL = "factual" # "What is the IUCN status of lions?" → SQL LOOKUP = "lookup" # "What CE rating does predator removal get?" → DB direct SYNTHESIS = "synthesis" # "What works for seabirds in NZ?" → full RAG def classify_query(question: str) -> QueryType: q = question.lower() # Pattern matching — simple and reliable if any(w in q for w in ["iucn status", "red list", "extinct", "endemic"]): return QueryType.FACTUAL if any(w in q for w in ["ce rating", "effectiveness score", "what score"]): return QueryType.LOOKUP return QueryType.SYNTHESIS def route_query(question: str, context) -> EvidenceSynthesis: qtype = classify_query(question) if qtype == QueryType.FACTUAL: return handle_factual(question, context) if qtype == QueryType.LOOKUP: return handle_lookup(question, context) return handle_synthesis(question, context) # full RAG only if needed

QUESTION: {question} SPECIES / HABITAT: {context} LOCATION: {location} RETRIEVED EVIDENCE (cite by [N]): {evidence_chunks} CE DATABASE RATINGS: {ce_ratings} REQUIRED OUTPUT — return ONLY this JSON structure: {{ "answer": "2-4 sentence direct answer", "confidence": "strong|moderate|weak|none", "interventions": [ {{"name": "...", "ce_category": "...", "effectiveness_pct": 75, "n_studies": 12, "outcome_direction": "positive"}} ], "evidence_gaps": ["list", "of", "gaps"], "citations": [{{"index":1,"paper_id":"...","title":"...","year":2023,"source":"CORE"}}], "geo_limits": "note if evidence is geographically limited or null", "taxa_limits": "note if evidence is taxa-specific or null" }} NEVER invent facts. Cite ONLY from provided evidence. If evidence is absent, set confidence to 'none'.""" def synthesise(question, context, location, chunks, ce_rows, max_retries=3): evidence_text = "\n\n".join( [f"[{i+1}] {c.page_content} ({c.metadata.get('year','?')})" for i, c in enumerate(chunks[:8])]) prompt = SYNTHESIS_PROMPT.format( question=question, context=context, location=location, evidence_chunks=evidence_text, ce_ratings=format_ce(ce_rows)) for attempt in range(max_retries): try: r = requests.post("http://localhost:11434/api/generate", json={ "model": "gemma4:e4b", "prompt": prompt, "stream": False, "format": "json", "options": {"temperature": 0.05} # near-deterministic }, timeout=60) raw = json.loads(r.json()['response']) return EvidenceSynthesis(**raw) # Pydantic validates — raises if malformed except Exception as e: if attempt == max_retries - 1: return safe_fallback_response(question, chunks) # never fail silently
